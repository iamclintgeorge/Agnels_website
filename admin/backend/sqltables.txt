--TRAINING PLACEMENT
-- Overview Table
CREATE TABLE placement_overview (
    id INT PRIMARY KEY AUTO_INCREMENT,
    description TEXT,
    vision TEXT,
    mission TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Statistics Table
CREATE TABLE placement_statistics (
    id INT PRIMARY KEY AUTO_INCREMENT,
    academic_year VARCHAR(255) NOT NULL,
    total_placements INT,
    average_package DECIMAL(10,2),
    highest_package DECIMAL(10,2),
    companies_visited INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Recruiters Table
CREATE TABLE placement_recruiters (
    id INT PRIMARY KEY AUTO_INCREMENT,
    company_name VARCHAR(255) NOT NULL,
    description TEXT,
    logo_path VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

--ANNOUNCEMENTS
CREATE TABLE `announcements` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `subject` varchar(120) NOT NULL,
  `description` text NOT NULL,
  `attachment` varchar(1000) DEFAULT NULL,
  `created_by` int(1) NOT NULL,
  `created_timestamp` timestamp NOT NULL DEFAULT current_timestamp() ON UPDATE current_timestamp(),
  `deleted` int(1) NOT NULL DEFAULT 0,
  PRIMARY KEY (`id`)
)
